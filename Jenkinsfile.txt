pipeline {
    agent any

    environment {
        IMAGE_NAME = 'cinir21/spring-petclinic:latest'
        CONTAINER_NAME = 'spring-petclinic'
        GRAFANA_CONTAINER_NAME = 'grafana'
        PROMETHEUS_CONTAINER_NAME = 'prometheus'
    }

    stages {
        stage('Deploy Prometheus and Grafana') {
            steps {
                script {
                    echo "Deploying Prometheus..."
                    bat '''
                    docker run -d --name=%PROMETHEUS_CONTAINER_NAME% -p 9090:9090 prom/prometheus:latest
                    '''

                    echo "Deploying Grafana..."
                    bat '''
                    docker run -d --name=%GRAFANA_CONTAINER_NAME% -p 3000:3000 grafana/grafana:latest
                    '''
                }
            }
        }

        stage('Deploy Application') {
            steps {
                bat '''
                docker stop %CONTAINER_NAME% || echo "No container to stop"
                docker rm %CONTAINER_NAME% || echo "No container to remove"
                docker pull %IMAGE_NAME%
                docker run -d --name %CONTAINER_NAME% -p 8086:8080 %IMAGE_NAME%
                '''
            }
        }
    }
}
