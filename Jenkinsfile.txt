pipeline {
    agent any

    environment {
        IMAGE_NAME = 'cinir21/spring-petclinic:latest'
        CONTAINER_NAME = 'spring-petclinic'
    }

    stages {
        stage('Maven Install') {
            steps {
                bat '''
                docker run --rm -v "%CD%":/app -w /app maven:3.8.7 mvn clean install
                '''
            }
        }

        stage('Docker Build') {
            steps {
                bat 'docker build -t %IMAGE_NAME% .'
            }
        }

        stage('Docker Push') {
            steps {
                withCredentials([usernamePassword(credentialsId: 'docker-hub-credentials', passwordVariable: 'dockerHubPassword', usernameVariable: 'dockerHubUser')]) {
                    bat '''
                    echo %dockerHubPassword% | docker login -u %dockerHubUser% --password-stdin
                    docker push %IMAGE_NAME%
                    '''
                }
            }
        }

        stage('Deploy Container') {
            steps {
                bat '''
                docker stop %CONTAINER_NAME% || echo "No container to stop"
                docker rm %CONTAINER_NAME% || echo "No container to remove"
                docker pull %IMAGE_NAME%
                docker run -d --name %CONTAINER_NAME% -p 8086:8080 %IMAGE_NAME%
                '''
            }
        }
    }
}
